Проверка частей часа через if-else

Давайте определим, в какую четверть часа попадает указанное количество минут:

let min = 10;

if (min >= 0 && min <= 14) {
console.log('1 четверть'); }

if (min >= 15 && min <= 29) {
console.log('2 четверть'); }

if (min >= 30 && min <= 44) {
console.log('3 четверть'); }

if (min >= 45 && min <= 59) {
console.log('4 четверть'); }

Задача 1: Решите аналогичную задачу, только определите в какую треть часа попадает указанное количество минут.

Давайте напишем условие, которое выведет сообщение в консоль, если длина строки равна или больше трех символов:

let str = '12345';

if (str.length >= 3) {
console.log('!');
}

Проверка символов строки

Давайте проверим, равен ли первый символ этой строки числу 1:

let str = '12345';

if (str[0] == 1) {
console.log('!');
}

Проверим теперь последний символ на равенство числу 5:

let str = '12345';
let last = str[str.length - 1];

if (last == 5) {
console.log('!');
}

console.log('делится с остатком ' + rest);
}
